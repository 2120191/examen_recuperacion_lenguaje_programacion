def es_vocal(caracter):
    vocales = ['a', 'e', 'i', 'o', 'u']
    if caracter.lower() in vocales:
        return True
    else:
        return False

# Ejemplo de uso
caracter = input("Ingresa un car√°cter: ")
resultado = es_vocal(caracter)

if resultado:
    print(f"{caracter} es una vocal.")
else:
    print(f"{caracter} no es una vocal.")
